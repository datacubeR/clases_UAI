import torch
import torch.nn as nn
from torch.utils.data import Dataset, DataLoader


class CaliforniaDataset(Dataset):
    def __init__(self, X, y):
        self.X = torch.from_numpy(X.values).float()
        self.y = torch.from_numpy(y.astype("float32").values)

    def __len__(self):
        return len(self.X)

    def __getitem__(self, idx):
        return dict(X=self.X[idx], y=self.y[idx])


train_data = CaliforniaDataset(X_train, y_train)

model_hp = dict(
    input_dim=28 * 28,
    hidden_dim_1=128,
    hidden_dim_2=64,
    output_dim=10,
    dropout_p=0.2,
    batch_norm=True,
    xavier_init=None,
)
class MLP(nn.Module):
    def __init__(self, network_hp):
        super().__init__()
        self.network_hp = network_hp
        self.fc1 = nn.Linear(
            network_hp["input_dim"], network_hp["hidden_dim_1"]
        )
        self.fc2 = nn.Linear(
            network_hp["hidden_dim_1"], network_hp["hidden_dim_2"]
        )
        self.fc3 = nn.Linear(
            network_hp["hidden_dim_2"], network_hp["output_dim"]
        )
        self.relu = nn.ModuleList(
            [nn.ReLU(inplace=True) for _ in range(2)]
        )

        if self.network_hp["xavier_init"] == "normal":
            nn.init.xavier_normal_(self.fc1.weight)
            nn.init.xavier_normal_(self.fc2.weight)
            nn.init.xavier_normal_(self.fc3.weight)
        elif self.network_hp["xavier_init"] == "uniform":
            nn.init.xavier_uniform_(self.fc1.weight)
            nn.init.xavier_uniform_(self.fc2.weight)
            nn.init.xavier_uniform_(self.fc3.weight)

        if self.network_hp["dropout_p"] is not None:
            self.dropout = nn.ModuleList(
                [
                    nn.Dropout(self.network_hp["dropout_p"])
                    for _ in range(2)
                ]
            )

        if self.network_hp["batch_norm"]:
            self.bn = nn.ModuleList(
                [
                    nn.BatchNorm1d(network_hp[f"hidden_dim_{i+1}"])
                    for i in range(2)
                ]
            )

    def forward(self, x):
        x = self.dense_block(
            x,
            self.fc1,
            relu_idx=0,
            bn_idx=0 if self.network_hp["batch_norm"] else None,
            drop_idx=(
                0 if self.network_hp["dropout_p"] is not None else None
            ),
        )
        x = self.dense_block(
            x,
            self.fc2,
            relu_idx=1,
            bn_idx=1 if self.network_hp["batch_norm"] else None,
            drop_idx=(
                1 if self.network_hp["dropout_p"] is not None else None
            ),
        )
        x = self.fc3(x)
        return x

    def dense_block(
        self, x, fc, relu_idx=None, bn_idx=None, drop_idx=None
    ):
        x = fc(x)
        if bn_idx is not None:
            x = self.bn[bn_idx](x)
        if relu_idx is not None:
            x = self.relu[relu_idx](x)
        if drop_idx is not None:
            x = self.dropout[drop_idx](x)
        return x

def train_model(
    train_data,
    val_data,
    model,
    criterion,
    optimizer,
    training_params,
    hardware_params,
):
    model.to(hardware_params["device"])
    optimizer = torch.optim.Adam(
        model.parameters(),
        lr=training_params["learning_rate"],
        weight_decay=training_params["weight_decay"],
    )

    train_dataloader = DataLoader(
        train_data,
        batch_size=training_params["batch_size"],
        shuffle=True,
        num_workers=hardware_params["num_workers"],
        pin_memory=hardware_params["pin_memory"],
    )
    val_dataloader = DataLoader(
        val_data,
        batch_size=training_params["batch_size"],
        shuffle=False,
        num_workers=hardware_params["num_workers"],
        pin_memory=hardware_params["pin_memory"],
    )

    for e in range(training_params["num_epochs"]):
        model.train()
        for batch in train_dataloader:
            X, y = batch["X"].to(hardware_params["device"]), batch["y"].to(
                hardware_params["device"]
            )

            optimizer.zero_grad()
            y_hat = model(X)
            loss = criterion(y_hat, y)
            loss.backward()
            optimizer.step()

        model.eval()
        with torch.no_grad():
            for batch in val_dataloader:
                X, y = batch["X"].to(hardware_params["device"]), batch[
                    "y"
                ].to(hardware_params["device"])
                y_hat = model(X)
                val_loss = criterion(y_hat, y)
