from sklearn.neighbors import KNeighborsClassifier


def knn(X, y, k=3):
    knn = KNeighborsClassifier(n_neighbors=k, n_jobs=-1)
    knn.fit(X, y)
    print(f"Puntaje para k = {k}: {knn.score(X, y)}")
